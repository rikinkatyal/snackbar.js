snackbar.js

A simple jQuery plugin that replicates Google's Material Design snackbar


$("body").snackbar();

Message Only


$("body").snackbar({
	message: "Message sent successfully"
});

Message with Option


$("body").snackbar({
	message: "Message archived",
	option: true,
	optionText: "UNDO",
	callback: function() {
		// Run code if user clicks on option
	}
});

Really Long Message


$("body").snackbar({
	message: "Lorem ipsum dolor sit...",
	duration: 20
});

Swipe to Dismiss


$("body").snackbar({
	message: "Swipe to dismiss me!",
	swipe: true
});

Custom Colors


$("body").snackbar({
	message: "I can be any color you want!",
	option: true,
	optionText: "HERE TOO!",
	primaryCol: "#FF69B4",
	accentCol: "#00FFFF"
});
